def ensure_pubsub
  return if defined? @pubsub
  @pubsub = redis.pubsub
  @pubsub_errbacks = []
  @pubsub.on(:disconnected, &method(:on_redis_close))
  EM::Synchrony.sync(@pubsub)
end

def pubsub(&on_disconnected)
  ensure_pubsub
  return false unless @pubsub.connected?
  @pubsub_errbacks << on_disconnected
  true
end
