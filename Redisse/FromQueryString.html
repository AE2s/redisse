def channels(env)
  query_string = env['QUERY_STRING'] || ''
  channels = query_string.split('&').map { |channel|
    URI.decode_www_form_component(channel)
  }
  channels.delete('polling')
  channels.delete_if {|channel| channel.start_with?('lastEventId=') }
end
